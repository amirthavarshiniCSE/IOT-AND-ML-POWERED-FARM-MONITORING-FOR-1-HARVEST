void setup() 
{
  Serial.begin(9600);
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass); // Connects to WiFi and Blynk
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("connect wifi:");
  lcd.setCursor(0, 1);
  lcd.print("waiting.........");
  dustSensor.begin();

  pinMode(pump_water, OUTPUT);
  pinMode(pump_distribute, OUTPUT);
  pinMode(pump_chemical, OUTPUT);
  pinMode(solinoid1, OUTPUT);
  pinMode(solinoid2, OUTPUT);
  pinMode(sw_pin, INPUT_PULLUP);

  digitalWrite(pump_water, LOW);
  digitalWrite(pump_distribute, LOW);
  digitalWrite(pump_chemical, LOW);
  digitalWrite(solinoid1, LOW);
  digitalWrite(solinoid2, LOW);
  delay(1000);
  lcd.clear();

}

void WriteIntToLCD4digit(unsigned int dat)
{
  lcd.print(dat / 1000);
  lcd.print((dat % 1000) / 100);
  lcd.print(((dat % 1000) % 100) / 10);
  lcd.print(((dat % 1000) % 100) % 10);
}
void WriteIntToLCD3digit(unsigned int dat)
{
  lcd.print(dat / 100);
  lcd.print((dat % 100) / 10);
  lcd.print((dat % 100) % 10);
}
void WriteIntToLCD2digit(unsigned int dat)
{
  lcd.print(dat / 10);
  lcd.print(dat % 10);
}

void serialEvent() 
{
  while (Serial.available()) 
  {
    char inChar = (char)Serial.read();
    
    rx[x]= inChar;
    if(rx[0]=='P')
    {
      x++;
    }
    else if(rx[0]!='P')
    {
      x=0;
    }
    if(x>30)x=30;
  }
}
